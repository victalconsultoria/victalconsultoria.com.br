    <h3>Buscar Produto</h3>

    <input type="text" id="busca" placeholder="Digite código ou nome..." onkeyup="buscarProduto()">

    <div id="resultadoBusca"></div>

</div>

    <h3>Lista de Cotação</h3>

    <div id="listaCarrinho"></div>

    <button onclick="gerarPDF()">Gerar PDF / Enviar Orçamento</button>

</div>

<script>

    let baseDados = [];

    let carrinho = [];

    // URL do CSV Publicado do Google Drive (Substituir pelo teu link)

    const urlCSV = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vR2ZD5mHtSutLQKXCW5Ps1zG-EHdRs_zFdBBjOHR9VRK9Ag57iNbAFsb7799stO1w/pubhtml';

    // Função para carregar dados

    function carregarDados() {

        Papa.parse(urlCSV, {

            download: true,

            header: true,

            complete: function(results) {

                baseDados = results.data;

                alert("Base de dados atualizada!");

            }

        });

    }

    // Busca de Produto

    function buscarProduto() {

        const termo = document.getElementById('busca').value.toLowerCase();

        const lista = document.getElementById('resultadoBusca');

        lista.innerHTML = "";

        if(termo.length < 2) return;

        const filtrados = baseDados.filter(p =>

            p.Descricao.toLowerCase().includes(termo) || p.Codigo.includes(termo)

        ).slice(0, 5);

        filtrados.forEach(p => {

            const div = document.createElement('div');

            div.className = 'produto-item';

            div.innerHTML = `

                <strong>${p.Codigo} - ${p.Descricao}</strong><br>

                Preço NF: R$ ${p.PrecoNF} | <button onclick='addCarrinho("${p.Codigo}")'>Adicionar</button>

            `;

            lista.appendChild(div);

        });

    }

    function addCarrinho(codigo) {

        const prod = baseDados.find(p => p.Codigo === codigo);

        const perfil = document.getElementById('perfilFiscal').value;

       

        if(!perfil) return alert("Selecione o perfil fiscal primeiro!");

        // Aqui o sistema busca o preço na coluna correspondente ao perfil

        const precoCalculado = prod[perfil] || prod.PrecoNF;

        carrinho.push({...prod, precoFinal: precoCalculado});

        atualizarCarrinho();

    }

    function atualizarCarrinho() {

        const lista = document.getElementById('listaCarrinho');

        lista.innerHTML = carrinho.map(i => `<div>${i.Descricao} - R$ ${i.precoFinal}</div>`).join('');

    }

    async function gerarPDF() {

        const { jsPDF } = window.jspdf;

        const doc = new jsPDF();

       

        doc.text("Orçamento de Materiais", 10, 10);

        doc.text(`Cliente: ${document.getElementById('razao').value}`, 10, 20);

       

        let y = 30;

        carrinho.forEach(item => {

            doc.text(`${item.Codigo} - ${item.Descricao}: R$ ${item.precoFinal}`, 10, y);

            y += 10;

        });

        doc.save("orcamento.pdf");

        alert("PDF Gerado! Agora pode partilhar via WhatsApp.");

    }

    // Iniciar carga

    window.onload = carregarDados;

</script>
